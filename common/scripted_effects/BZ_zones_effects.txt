@BRYANSK_REGION = 0
init_create_bz_unit = {
	set_temp_variable = { used_manpower = bz_selected_states^num }
	add_to_temp_variable = { used_manpower = 2 }
	if = {
		limit = {
			has_country_flag = BZ_choose_offensive_bryansk
		}
		clamp_temp_variable = { var = used_manpower min = 1 max = 7 }
		divide_temp_variable = { used_manpower = 7 }
		multiply_temp_variable = { used_manpower = BZ_manpower_by_states^@BRYANSK_REGION }
	}
}
